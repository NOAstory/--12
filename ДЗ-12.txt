using System;

class Program
{
    static void Main()
    {
        const string CommandHello = "hello";
        const string CommandInfo = "info";
        const string CommandRandom = "random";
        const string CommandClear = "clear";
        const string CommandExit = "exit";

        string availableCommands = $"\nКоманды: {CommandHello}, {CommandInfo}, {CommandRandom}, {CommandClear}, {CommandExit}";
        Random random = new Random();
        int minRandomValue = 1;
        int maxRandomValue = 100;

        bool isRunning = true;

        while (isRunning)
        {
            Console.WriteLine(availableCommands);
            Console.Write("Введите команду: ");
            string command = Console.ReadLine();

            switch (command)
            {
                case CommandHello:
                    Console.WriteLine("Привет! Это простая программа.");
                    break;

                case CommandInfo:
                    Console.WriteLine("Информация: это программа для обработки команд.");
                    break;

                case CommandRandom:
                    int randomNumber = random.Next(minRandomValue, maxRandomValue + 1);
                    Console.WriteLine($"Случайное число: {randomNumber}");
                    break;

                case CommandClear:
                    Console.Clear();
                    break;

                case CommandExit:
                    Console.WriteLine("Выход из программы.");
                    isRunning = false;
                    break;

                default:
                    Console.WriteLine("Неизвестная команда. Попробуйте еще раз.");
                    break;
            }
        }
    }
}
